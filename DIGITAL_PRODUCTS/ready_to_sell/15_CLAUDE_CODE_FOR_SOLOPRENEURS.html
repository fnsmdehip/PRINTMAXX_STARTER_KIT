The Overnight Loop (Ralph Pattern)

While you sleep, your agents work. This is the pattern that makes it possible.

The Command

while :; do
  cat PROMPT.md | claude --dangerously-skip-permissions --print
done

That is the entire loop. The sophistication is in how PROMPT.md is structured and what state files exist on disk.

Why this works: Each iteration gets a fresh 200K context window. No degradation. No crashes. The filesystem is the memory. Git is the version control. The bash loop is the scheduler. Every iteration reads state from files, does exactly one task, writes results back, and exits cleanly.

  • Publishing to your main website without review

Website Audit Script (Core Logic)

import requests
from bs4 import BeautifulSoup

def audit_site(url: str) -> dict:
    """Find specific problems on a prospect's site."""
    try:
        resp = requests.get(url, timeout=10, headers={"User-Agent": "Mozilla/5.0"})
        soup = BeautifulSoup(resp.text, "html.parser")
        text = soup.get_text(separator=" ", strip=True).lower()

        problems = []

        # Check for missing trust signals
        if not any(w in text for w in ["testimonial", "review", "case study", "client"]):
            problems.append("no social proof visible")

        # Check for vague value prop
        title = soup.title.string if soup.title else ""
        if len(title) < 20 or "welcome" in title.lower():
            problems.append("unclear homepage value prop")

        # Check for missing CTA
        ctas = soup.find_all(["button", "a"], string=lambda t: t and
               any(w in t.lower() for w in ["book", "schedule", "contact", "get started"]))
        if len(ctas) < 2:
            problems.append("weak or missing call-to-action")

        # Check for slow load signals (image-heavy)
        images = soup.find_all("img")
        if len(images) > 20:
            problems.append("likely slow load time from unoptimized images")

        return {"url": url, "problems": problems, "top_problem": problems[0] if problems else None}

    except Exception as e:
        return {"url": url, "problems": [], "error": str(e)}

Template 5: Basic Agent Python Script

#!/usr/bin/env python3
"""Template: Autonomous solopreneur agent with state persistence."""
import json
from pathlib import Path
from datetime import datetime

PROJECT_ROOT = Path(__file__).resolve().parent.parent
STATE_FILE = PROJECT_ROOT / "agents" / "my_agent" / "state.json"
LOG_FILE = PROJECT_ROOT / "logs" / "my_agent.log"

def load_state() -> dict:
    if STATE_FILE.exists():
        return json.loads(STATE_FILE.read_text())
    return {"cycles": 0, "last_run": None, "results": []}

def save_state(state: dict) -> None:
    state["last_run"] = datetime.now().isoformat()
    state["cycles"] += 1
    STATE_FILE.write_text(json.dumps(state, indent=2))

def log(msg: str) -> None:
    with open(LOG_FILE, "a") as f:
        f.write(f"[{datetime.now().isoformat()}] {msg}\n")

def main() -> None:
    state = load_state()
    log(f"Cycle {state['cycles'] + 1} starting")

    # 1. Read inputs (CSV, API, scraped data)
    # 2. Process / score / decide
    # 3. Write outputs (files, queues, state)
    # 4. Log result

    result = {"action": "example", "status": "success", "count": 0}
    state["results"].append(result)
    save_state(state)
    log(f"Done: {result}")

if __name__ == "__main__":
    main()
